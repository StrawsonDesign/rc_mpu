1 /*******************************************************************************
2 * Linear Algebra
3 *
4 * @ int rc_matrix_times_col_vec(rc_matrix_t A, rc_vector_t v, rc_vector_t* c)
5 *
6 * Multiplies matrix A times column vector v and places the result in column
7 * vector c. Any existing data in c is freed if necessary and c is resized
8 * appropriately. Vectors v and c are interpreted as column vectors, but nowhere
9 * in their definitions are they actually specified as one or the other.
10 * Returns 0 on success and -1 on failure.
11 *
12 * @ int rc_row_vec_times_matrix(rc_vector_t v, rc_matrix_t A, rc_vector_t* c)
13 *
14 * Multiplies row vector v times matrix A and places the result in row
15 * vector c. Any existing data in c is freed if necessary and c is resized
16 * appropriately. Vectors v and c are interpreted as row vectors, but nowhere
17 * in their definitions are they actually specified as one or the other.
18 * Returns 0 on success and -1 on failure.
19 *
20 * @ float rc_matrix_determinant(rc_matrix_t A)
21 *
22 * Returns the determinant of square matrix A or -1.0f on failure.
23 *
24 * @ int rc_lup_decomp(rc_matrix_t A, rc_matrix_t* L, rc_matrix_t* U, rc_matrix_t* P)
25 *
26 * Performs LUP decomposition on matrix A with partial pivoting and places the
27 * result in matrices L,U,&P. Matrix A remains untouched and the original
28 * contents of LUP (if any) are freed and LUP are resized appropriately.
29 * Returns 0 on success or -1 on failure.
30 *
31 * @ int rc_qr_decomp(rc_matrix_t A, rc_matrix_t* Q, rc_matrix_t* R)
32 *
33 * Uses householder reflection method to find the QR decomposition of A.
34 * Returns 0 on success or -1 on failure.
35 *
36 * @ int rc_invert_matrix(rc_matrix_t A, rc_matrix_t* Ainv)
37 *
38 * Inverts Matrix A via LUP decomposition method and places the result in matrix
39 * Ainv. Any existing memory allocated for Ainv is freed if necessary and its
40 * contents are overwritten. Returns 0 on success or -1 on failure such as if
41 * matrix A is not invertible.
42 *
43 * @ int rc_invert_matrix_inplace(rc_matrix_t* A)
44 *
45 * Inverts Matrix A in place. The original contents of A are lost.
46 * Returns 0 on success or -1 on failure such as if A is not invertible.
47 *
48 * @ int rc_lin_system_solve(rc_matrix_t A, rc_vector_t b, rc_vector_t* x)
49 *
50 * Solves Ax=b for given matrix A and vector b. Places the result in vector x.
51 * existing contents of x are freed and new memory is allocated if necessary.
52 * Thank you to Henry Guennadi Levkin for open sourcing this routine, it's
53 * adapted here for RC use and includes better detection of unsolvable systems.
54 * Returns 0 on success or -1 on failure.
55 *
56 * @ int rc_lin_system_solve_qr(rc_matrix_t A, rc_vector_t b, rc_vector_t* x)
57 *
58 * Finds a least-squares solution to the system Ax=b for non-square A using QR
59 * decomposition method and places the solution in x.
60 * Returns 0 on success or -1 on failure.
61 *
62 * @ int rc_fit_ellipsoid(rc_matrix_t pts, rc_vector_t* ctr, rc_vector_t* lens)
63 *
64 * Fits an ellipsoid to a set of points in 3D space. The principle axes of the
65 * fitted ellipsoid align with the global coordinate system. Therefore there are
66 * 6 degrees of freedom defining the ellipsoid: the x,y,z coordinates of the
67 * centroid and the lengths from the centroid to the surface in each of the 3
68 * directions.
69 *
70 * rc_matrix_t 'pts' is a tall matrix with 3 columns and at least 6 rows.
71 * Each row must contain the x,y&z components of each individual point to be fit.
72 * If only 6 rows are provided, the resulting ellipsoid will be an exact fit.
73 * Otherwise the result is a least-squares fit to the over-defined dataset.
74 *
75 * The final x,y,z position of the centroid will be placed in vector 'ctr' and
76 * the lengths or radius from the centroid to the surface along each axis will
77 * be placed in the vector 'lens'
78 *
79 * Returns 0 on success or -1 on failure.
80 *******************************************************************************/
