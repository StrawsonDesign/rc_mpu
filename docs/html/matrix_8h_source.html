1 /*******************************************************************************
2 * Matrix
3 *
4 * @ int rc_alloc_matrix(rc_matrix_t* A, int rows, int cols)
5 *
6 * Allocates memory for matrix A to have new dimensions given by arguments rows
7 * and cols. If A is initially the right size, nothing is done and the data in A
8 * is preserved. If A is uninitialized or of the wrong size then any existing
9 * memory is freed and new memory is allocated, helping to prevent accidental
10 * memory leaks. The contents of the new matrix is not guaranteed to be anything
11 * in particular.
12 * Returns 0 on success, otherwise -1. Will only be unsuccessful if
13 * rows&cols are invalid or there is insufficient memory available.
14 *
15 * @ int rc_free_matrix(rc_matrix_t* A)
16 *
17 * Frees the memory allocated for a matrix A and importantly sets the dimensions
18 * and initialized flag of the rc_matrix_t struct to 0 to indicate to other
19 * functions that A no longer points to allocated memory and cannot be used until
20 * more memory is allocated such as with rc_alloc_matrix or rc_matrix_zeros.
21 * Returns 0 on success. Will only fail and return -1 if it is passed a NULL
22 * pointer.
23 *
24 * @ rc_matrix_t rc_empty_matrix()
25 *
26 * Returns an rc_matrix_t with no allocated memory and the initialized flag set
27 * to 0. This is useful for initializing rc_matrix_t structs when they are
28 * declared since local variables declared in a function without global variable
29 * scope in C are not guaranteed to be zeroed out which can lead to bad memory
30 * pointers and segfaults if not handled carefully. We recommend initializing all
31 * matrices with this before using rc_alloc_matrix or any other function.
32 *
33 * @ int rc_matrix_zeros(rc_matrix_t* A, int rows, int cols)
34 *
35 * Resizes matrix A and allocates memory for a matrix with specified rows &
36 * columns. The new memory is pre-filled with zeros. Any existing memory
37 * allocated for A is freed if necessary to avoid memory leaks.
38 * Returns 0 on success or -1 on error.
39 *
40 * @ int rc_identity_matrix(rc_matrix_t* A, int dim)
41 *
42 * Resizes A to be a square identity matrix with dimensions dim-by-dim. Any
43 * existing memory allocated for A is freed if necessary to avoid memory leaks.
44 * Returns 0 on success or -1 on failure.
45 *
46 * @ int rc_random_matrix(rc_matrix_t* A, int rows, int cols)
47 *
48 * Resizes A to be a matrix with the specified number of rows and columns and
49 * populates the new memory with random numbers evenly distributed between -1.0
50 * and 1.0. Any existing memory allocated for A is freed if necessary to avoid
51 * memory leaks. Returns 0 on success or -1 on failure.
52 *
53 * @ int rc_diag_matrix(rc_matrix_t* A, rc_vector_t v)
54 *
55 * Resizes A to be a square matrix with the same number of rows and columns as
56 * vector v's length. The diagonal entries of A are then populated with the
57 * contents of v and the off-diagonal entries are set to 0. The original contents
58 * of A are freed to avoid memory leaks.
59 * Returns 0 on success or -1 on failure.
60 *
61 * @ int rc_duplicate_matrix(rc_matrix_t A, rc_matrix_t* B)
62 *
63 * Makes a duplicate of the data from matrix A and places into matrix B. If B is
64 * already the right size then its contents are overwritten. If B is unallocated
65 * or is of the wrong size then the memory is freed if necessary and new memory
66 * is allocated to hold the duplicate of A.
67 * Returns 0 on success or -1 on error.
68 *
69 * @ int rc_set_matrix_entry(rc_matrix_t* A, int row, int col, float val)
70 *
71 * Sets the specified single entry of matrix A to 'val' where the position is
72 * zero-indexed at the top-left corner. In practice this is never used as it is
73 * much easier for the user to set values directly with this code:
74 *
75 * A.d[row][col]=val;
76 *
77 * However, we provide this function for completeness. It is not strictly
78 * necessary for A to be provided as a pointer since a copy of the struct A
79 * would also contain the correct pointer to the original matrix's allocated
80 * memory. However, in this library we use the convention of passing an
81 * rc_vector_t struct or rc_matrix_struct as a pointer when its data is to be
82 * modified by the function, and as a normal argument when it is only to be read
83 * by the function. Returns 0 on success or -1 on error.
84 *
85 * @ float rc_get_matrix_entry(rc_matrix_t A, int row, int col)
86 *
87 * Returns the specified single entry of matrix 'A' in position 'pos' where the
88 * position is zero-indexed. Returns -1.0f on failure and prints an error message
89 * to stderr. In practice this is never used as it is much easier for the user to
90 * read values directly with this code:
91 *
92 * val = A.d[row][col];
93 *
94 * However, we provide this function for completeness. It also provides sanity
95 * checks to avoid possible segfaults.
96 *
97 * @ int rc_print_matrix(rc_matrix_t A)
98 *
99 * Prints the contents of matrix A to stdout in decimal notation with 4 decimal
100 * places. Not recommended for very large matrices as rows will typically
101 * linewrap if the terminal window is not wide enough.
102 *
103 * @ void rc_print_matrix_sci(rc_matrix_t A)
104 *
105 * Prints the contents of matrix A to stdout in scientific notation with 4
106 * significant figures. Not recommended for very large matrices as rows will
107 * typically linewrap if the terminal window is not wide enough.
108 *
109 * @ int rc_matrix_times_scalar(rc_matrix_t* A, float s)
110 *
111 * Multiplies every entry in A by scalar value s. It is not strictly
112 * necessary for A to be provided as a pointer since a copy of the struct A
113 * would also contain the correct pointer to the original matrix's allocated
114 * memory. However, in this library we use the convention of passing an
115 * rc_vector_t struct or rc_matrix_struct as a pointer when its data is to be
116 * modified by the function, and as a normal argument when it is only to be read
117 * by the function. Returns 0 on success or -1 on failure.
118 *
119 * @ int rc_multiply_matrices(rc_matrix_t A, rc_matrix_t B, rc_matrix_t* C)
120 *
121 * Multiplies A*B=C. C is resized and its original contents are freed if
122 * necessary to avoid memory leaks. Returns 0 on success or -1 on failure.
123 *
124 * @ int rc_left_multiply_matrix_inplace(rc_matrix_t A, rc_matrix_t* B)
125 *
126 * Multiplies A*B and puts the result back in the place of B. B is resized and
127 * its original contents are freed if necessary to avoid memory leaks.
128 * Returns 0 on success or -1 on failure.
129 *
130 * @ int rc_right_multiply_matrix_inplace(rc_matrix_t* A, rc_matrix_t B)
131 *
132 * Multiplies A*B and puts the result back in the place of A. A is resized and
133 * its original contents are freed if necessary to avoid memory leaks.
134 * Returns 0 on success or -1 on failure.
135 *
136 * @ int rc_add_matrices(rc_matrix_t A, rc_matrix_t B, rc_matrix_t* C)
137 *
138 * Resizes matrix C and places the sum A+B in C. The original contents of C are
139 * safely freed if necessary to avoid memory leaks.
140 * Returns 0 on success or -1 on failure.
141 *
142 * @ int rc_add_matrices_inplace(rc_matrix_t* A, rc_matrix_t B)
143 *
144 * Adds matrix B to A and places the result in A so the original contents of A
145 * are lost. Use rc_add_matrices if you wish to keep the contents of both matrix
146 * A and B. Returns 0 on success or -1 on failure.
147 *
148 * @ int rc_matrix_transpose(rc_matrix_t A, rc_matrix_t* T)
149 *
150 * Resizes matrix T to hold the transposed contents of A and leaves A untouched.
151 * Returns 0 on success or -1 on failure.
152 *
153 * @ int rc_matrix_transpose_inplace(rc_matrix_t* A)
154 *
155 * Transposes matrix A in place. Use as an alternative to rc_matrix_transpose
156 * if you no longer have need for the original contents of matrix A.
157 * Returns 0 on success or -1 on failure.
158 *******************************************************************************/
