1 /*******************************************************************************
2 * polynomial Manipulation
3 *
4 * We represent polynomials as a vector of coefficients with the highest power
5 * term on the left at vector index 0. The following polynomial manipulation
6 * functions are designed to behave like their counterparts in the Numerical
7 * Renaissance codebase.
8 *
9 * @ int rc_print_poly(rc_vector_t v)
10 *
11 * Like rc_print_vector, but assumes the contents represent a polynomial and
12 * prints the coefficients with trailing powers of x for easier reading. This
13 * relies on your terminal supporting unicode UTF-8.
14 *
15 * @ int rc_poly_conv(rc_vector_t a, rc_vector_t b, rc_vector_t* c)
16 *
17 * Convolutes the polynomials a&b and places the result in vector c. This finds
18 * the coefficients of the polynomials resulting from multiply a*b. The original
19 * contents of c are freed and new memory is allocated if necessary.
20 * returns 0 on success or -1 on failure.
21 *
22 * @ int rc_poly_power(rc_vector_t a, int n, rc_vector_t* b)
23 *
24 * Raises a polynomial a to itself n times where n is greater than or equal to 0.
25 * Places the result in vector b, any existing memory allocated for b is freed
26 * and its contents are lost. Returns 0 on success and -1 on failure.
27 *
28 * @ int rc_poly_add(rc_vector_t a, rc_vector_t b, rc_vector_t* c)
29 *
30 * Add two polynomials a&b with right justification and place the result in c.
31 * Any existing memory allocated for c is freed and its contents are lost.
32 * Returns 0 on success and -1 on failure.
33 *
34 * @ int rc_poly_add_inplace(rc_vector_t* a, rc_vector_t b)
35 *
36 * Adds polynomials b&a with right justification. The result is placed in vector
37 * a and a's original contents are lost. More memory is allocated for a if
38 * necessary. Returns 0 on success or -1 on failure.
39 *
40 * @ int rc_poly_subtract(rc_vector_t a, rc_vector_t b, rc_vector_t* c)
41 *
42 * Subtracts two polynomials a-b with right justification and place the result in
43 * c. Any existing memory allocated for c is freed and its contents are lost.
44 * Returns 0 on success and -1 on failure.
45 *
46 * @ int rc_poly_subtract_inplace(rc_vector_t* a, rc_vector_t b)
47 *
48 * Subtracts b from a with right justification. a stays in place and new memory
49 * is allocated only if b is longer than a.
50 *
51 * @ int rc_poly_differentiate(rc_vector_t a, int d, rc_vector_t* b)
52 *
53 * Calculates the dth derivative of the polynomial a and places the result in
54 * vector b. Returns 0 on success or -1 on failure.
55 *
56 * @ int rc_poly_divide(rc_vector_t n, rc_vector_t d, rc_vector_t* div, rc_vector_t* rem)
57 *
58 * Divides denominator d into numerator n. The remainder is placed into vector
59 * rem and the divisor is placed into vector div. Returns 0 on success or -1
60 * on failure.
61 *
62 * @ int rc_poly_butter(int N, float wc, rc_vector_t* b)
63 *
64 * Calculates vector of coefficients for continuous-time Butterworth polynomial
65 * of order N and cutoff wc (rad/s) and places them in vector b.
66 * Returns 0 on success or -1 on failure.
67 *******************************************************************************/
